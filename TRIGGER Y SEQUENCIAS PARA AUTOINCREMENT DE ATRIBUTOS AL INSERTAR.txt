CREAR SECUENCIAS Y TRIGGER PARA AUTO INCREMENTAR LAS ID AL INSERTAR DATOS:


SECUENCIA: 


CREATE SEQUENCE SEQ_CLIENTES <------ NOMBRE DE LA TABLA : POR BUENAS CONSTUMBRES SE USA EL PREFIJO SEQ DE SEQUENCE
Y LUEGO EL NOMBRE DE LA TABLA

START WITH 1 <---------- DESDE QUE NUMERO COMIENZA LA ID 


INCREMENT BY 1; <-------- DE CUANTO EN CUANTO SE INCREMENTA

EJEMPLO:
CREATE SEQUENCE SQ_CLIENTES
START WITH 1
INCREMENT BY 1;



TRIGGER:

CREATE TRIGGER TRIG_CLIENTES <------------ LO MISMO QUE LA SECUENCIA TRIG ES EL PREFIJO PARA EL TRIGGER Y LO DEMAS
BEFORE INSERT ON CLIENTES <------------------- IMPORTANTE: AQUI VA EL NOMBRE DE LA TABLA DONDE SE AUTO INCREMENTARA

FOR EACH ROW
BEGIN

SELECT SEQ_CLIENTES.NEXTVAL INTO :NEW.ID FROM DUAL; <---------- AQUI DESPUES DEL SELECT SE AGREGA EL NOMBRE DE LA SEQUENCIA
LUEGO DEL INTO SE AGREGA EL ATRIBUTO A INCREMENTAR QUE EN ESTE CASO SERIA LA ID.

END;

NOTA: SE DEBE CREAR UNA SEQUENCE Y UN TRIGGER POR CADA TABLA.!!
NOTA2: AL INSERTAR UN NUEVO VALOR, EL ATRIBUTO ID DEBE SER 0.
